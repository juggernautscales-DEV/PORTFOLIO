Backup Folder

Zip a target folder with a timestamp.

@echo off
color 0A
set SOURCE=C:\Projects\MyApp
set DATESTAMP=%DATE:~-4%%DATE:~4,2%%DATE:~7,2%
set TARGET=%USERPROFILE%\Backups\MyApp_%DATESTAMP%.zip
powershell -Command "Compress-Archive -Path '%SOURCE%' -DestinationPath '%TARGET%'"
echo Backup saved to %TARGET%
pause

Env Setup

Install tools and configure PATH.

@echo off
color 0A
echo === Env Setup ===
setx PATH "%PATH%;C:\Tools\bin"
choco install git -y
choco install nodejs -y
echo Done.
pause

Project Scaffolding

Create a new project directory with standard structure.

@echo off
color 0A
set NAME=%1
if "%NAME%"=="" set NAME=NewProject
mkdir %NAME%\src
mkdir %NAME%\tests
mkdir %NAME%\dist
echo Initialized %NAME%
pause
